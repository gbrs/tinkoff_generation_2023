        if (a <= b) {

        } else {
            v[i][0] = b;

        v[i + 1][1] = 99999;

    }

    b = 0;

            }

Задача F: Решаем контест

#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    long long n, a, sf, sd;
    sf = 0;
    sd = 0;
    std::cin >> n;
    std::vector<int> t(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a;
        t[i] = a;
        std::cin >> a;
        sd += a;
    }
    sort(t.begin(), t.end());
    for (int i = 0; i < n; i++) {
        sf += t[i] * (n - i);
    }
    std::cout << sd - sf;
}
